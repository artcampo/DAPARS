conditional flow
Prog  := STMTS

STMTS := STMT STMTS
      |  empty
STMT  := if(E){STMTS} IFELSE
      |  DECL;
      |  E = E;

IFELSE:= else{STMTS}
      |  empty

E     := T E'
E'    := + T E'
      |  empty
T     := F
F     := ( E )
      |  numerical

DECL  := TYPE NAME_LIST

TYPE  := int
      |  bool

NAME_LIST := name NAME_LIST
          | empty



NAME_LIST := name NAME_LIST'

NAME_LIST':= , name NAME_LIST'
          | empty
