conditional flow
Prog   := P_INIT STMTS P_END
P_INIT := empty
P_END  := empty

STMTS := STMT STMTS
      |  empty
STMT  := if(E){STMTS} IFELSE
      |  while(E){STMTS}
      |  DECL;
      |  E = E;

IFELSE:= else{STMTS}
      |  empty

E     := T E'
E'    := + T E'
      |  empty
T     := F
F     := ( E )
      |  numerical
      | true
      | false

DECL  := TYPE NAME_LIST

TYPE  := int
      |  bool

NAME_LIST := name NAME_LIST'

NAME_LIST':= , name NAME_LIST'
          | empty
